const API_GET_FOLDERS = "https://api.jasapromosi.web.id/api_get_folders.php"; const API_DELETE_FOLDER = "https://api.jasapromosi.web.id/api_delete_folder.php"; async function loadFolders() { const tbody = document.getElementById("target-list-folder"); // Pastikan ID ini sesuai dengan tbody indeks mas bro if(!tbody) return; try { const response = await fetch(API_GET_FOLDERS); const result = await response.json(); if (result.status === "success") { tbody.innerHTML = ""; if(result.data.length === 0) { tbody.innerHTML = `Belum ada folder.`; return; } result.data.forEach(folder => { tbody.innerHTML += ` ${folder.nama_folder} Dibuat: ${folder.created_at} ${folder.total_kontak} Nomor Kelola `; }); } } catch (error) { console.error("Gagal memuat folder", error); } } // Fungsi eksekusi hapus folder async function deleteFolder(idFolder) { if(!confirm("PERINGATAN! Menghapus folder ini juga akan MENGHAPUS SEMUA nomor WA di dalamnya. Yakin, Mas Bro?")) return; try { const response = await fetch(API_DELETE_FOLDER, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ folder_id: idFolder }) }); const result = await response.json(); if(result.status === "success") { loadFolders(); // Refresh daftar folder otomatis } else { alert("Gagal: " + result.message); } } catch (error) { alert("Gagal koneksi ke API Hapus Folder!"); } } // Jalankan saat halaman dibuka window.addEventListener("DOMContentLoaded", loadFolders);
